184  // Implementation has to be after numCells is defined
185  pmacc::traits::StringProperty getStringProperties( std::string const & name )
186  {
187  pmacc::traits::StringProperty propList;
188  const DataSpace<DIM3> periodic =
189  Environment<simDim>::get().EnvironmentController().getCommunicator().getPeriodic();
190 
191  for( uint32_t i = 1; i < NumberOfExchanges<simDim>::value; ++i )
192  {
193  // for each planar direction: left right top bottom back front
194  if( FRONT % i == 0 )
195  {
196  const std::string directionName = ExchangeTypeNames()[i];
197  const DataSpace<DIM3> relDir = Mask::getRelativeDirections<DIM3>(i);
198 
199  bool isPeriodic = false;
200  uint32_t axis = 0; // x(0) y(1) z(2)
201  uint32_t axisDir = 0; // negative (0), positive (1)
202  for( uint32_t d = 0; d < simDim; d++ )
203  {
204  if( relDir[d] * periodic[d] != 0 )
205  isPeriodic = true;
206  if( relDir[d] != 0 )
207  axis = d;
208  }
209  if( relDir[axis] > 0 )
210  axisDir = 1;
211 
212  std::string boundaryName = "open"; // absorbing boundary
213  if( isPeriodic )
214  boundaryName = "periodic";
215 
216  if( boundaryName == "open" )
217  {
218  std::ostringstream boundaryParam;
219  boundaryParam << name + " over "
220  << numCells[axis][axisDir] << " cells";
221  propList[directionName]["param"] = boundaryParam.str();
222  }
223  else
224  {
225  propList[directionName]["param"] = "none";
226  }
227 
228  propList[directionName]["name"] = boundaryName;
229  }
230  }
231  return propList;
232  }
