43 {
44 public:
45 
46  template<class BoxedMemory>
47  static void run(uint32_t currentStep, MappingDesc &cellDescription, BoxedMemory deviceBox)
48  {
49  const uint32_t numSlides = MovingWindow::getInstance().getSlideCounter(currentStep);
50  for (uint32_t i = 1; i < NumberOfExchanges<simDim>::value; ++i)
51  {
52  /* only call for planes: left right top bottom back front*/
53  if (FRONT % i == 0 && !(Environment<simDim>::get().GridController().getCommunicationMask().isSet(i)))
54  {
55  uint32_t direction = 0; /*set direction to X (default)*/
56  if (i >= BOTTOM && i <= TOP)
57  direction = 1; /*set direction to Y*/
58  if (i >= BACK)
59  direction = 2; /*set direction to Z*/
60 
61  /* exchange mod 2 to find positive or negative direction
62  * positive direction = 1
63  * negative direction = 0
64  */
65  uint32_t pos_or_neg = i % 2;
66 
67  uint32_t thickness = absorber::numCells[direction][pos_or_neg];
68  float_X absorber_strength = ABSORBER_STRENGTH[direction][pos_or_neg];
69 
70  if (thickness == 0) continue; /*if the absorber has no thickness we check the next side*/
71 
72  /* allow to enable the absorber on the top side if the laser
73  * initialization plane in y direction is *not* in cell zero
74  */
75  if (fields::laserProfiles::Selected::initPlaneY == 0)
76  {
77  /* disable the absorber on top side if
78  * no slide was performed and
79  * laser init time is not over
80  */
81  if (numSlides == 0 && ((currentStep * DELTA_T) <= fields::laserProfiles::Selected::INIT_TIME))
82  {
83  /* disable absorber on top side */
84  if (i == TOP) continue;
85  }
86  }
87 
88  /* if sliding window is active we disable absorber on bottom side*/
89  if (MovingWindow::getInstance().isSlidingWindowActive(currentStep) && i == BOTTOM) continue;
90 
91  ExchangeMapping<GUARD, MappingDesc> mapper(cellDescription, i);
92  constexpr uint32_t numWorkers = pmacc::traits::GetNumWorkers<
93  pmacc::math::CT::volume< SuperCellSize >::type::value
94  >::value;
95 
96  PMACC_KERNEL( KernelAbsorbBorder< numWorkers> {} )(
97  mapper.getGridDim(),
98  numWorkers
99  )(
100  deviceBox,
101  thickness,
102  absorber_strength,
103  mapper
104  );
105  }
106  }
107  }
108 
109 };
