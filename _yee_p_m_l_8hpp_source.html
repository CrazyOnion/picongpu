106  {
107  constexpr auto numWorkers = getNumWorkers( );
108  using Kernel = yeePML::KernelUpdateBHalf<
109  numWorkers,
110  BlockDescription< CurlE >
111  >;
112  AreaMapper< T_Area > mapper{ cellDescription };
113  /* Note: here it is possible to first check if PML is enabled
114  * in the local domain at all, and otherwise optimize by calling
115  * the normal Yee update kernel. We do not do that, as this
116  * would be fragile with respect to future separation of PML
117  * into a plugin.
118  */
119  PMACC_KERNEL( Kernel{ } )
120  ( mapper.getGridDim( ), numWorkers )(
121  mapper,
122  getLocalParameters( mapper, currentStep ),
123  CurlE( ),
124  fieldE->getDeviceDataBox( ),
125  fieldB->getDeviceDataBox( ),
126  psiB->getDeviceOuterLayerBox( )
127  );

269  Thickness getLocalThickness( uint32_t const currentStep ) const
270  {
271  /* The logic of the following checks is the same as in
272  * absorber::ExponentialDamping::run( ), to disable the absorber
273  * at a border we set the corresponding thickness to 0.
274  */
275  auto & movingWindow = MovingWindow::getInstance( );
276  auto const numSlides = movingWindow.getSlideCounter( currentStep );
277  auto const numExchanges = NumberOfExchanges< simDim >::value;
278  auto const communicationMask = Environment< simDim >::get( ).GridController( ).getCommunicationMask( );
279  Thickness localThickness = globalSize;
280  for( uint32_t exchange = 1u; exchange < numExchanges; ++exchange )
281  {
282  /* Here we are only interested in the positive and negative
283  * directions for x, y, z axes and not the "diagonal" ones.
284  * So skip other directions except left, right, top, bottom,
285  * back, front
286  */
287  if( FRONT % exchange != 0 )
288  continue;
289 
290  // Transform exchange into a pair of axis and direction
291  uint32_t axis = 0;
292  if( exchange >= BOTTOM && exchange <= TOP )
293  axis = 1;
294  if( exchange >= BACK )
295  axis = 2;
296  uint32_t direction = exchange % 2;
297 
298  // No PML at the borders between two local domains
299  bool hasNeighbour = communicationMask.isSet( exchange );
300  if( hasNeighbour )
301  localThickness( axis, direction ) = 0;
302 
303  // Disable PML during laser initialization
304  if( fields::laserProfiles::Selected::initPlaneY == 0 )
305  {
306  bool isLaserInitializationOver =
307  (currentStep * DELTA_T) >= fields::laserProfiles::Selected::INIT_TIME;
308  if( numSlides == 0 && !isLaserInitializationOver && exchange == TOP )
309  localThickness( axis, direction ) = 0;
310  }
311 
312  // Disable PML at the far side of the moving window
313  if( movingWindow.isSlidingWindowActive( currentStep ) && exchange == BOTTOM )
314  localThickness( axis, direction ) = 0;
315  }
316  return localThickness;
317  }

320  void checkLocalThickness( Thickness const localThickness ) const
321  {
322  auto const localDomain = Environment< simDim >::get( ).SubGrid( ).getLocalDomain( );
323  auto const localPMLSize = localThickness.negativeBorder + localThickness.positiveBorder;
324  auto pmlFitsDomain = true;
325  for( uint32_t dim = 0u; dim < simDim; dim++ )
326  if( localPMLSize[ dim ] > localDomain.size[ dim ] )
327  pmlFitsDomain = false;
328  if( !pmlFitsDomain )
329  throw std::out_of_range( "Requested PML size exceeds the local domain" );
330  }

395  void update_beforeCurrent( uint32_t const currentStep )
396  {
397  /* These steps are the same as in the Yee solver,
398  * PML updates are done as part of solver.updateE( ),
399  * solver.updateBHalf( )
400  */
401  solver.template updateBHalf < CORE + BORDER >( currentStep );
402  auto & fieldB = solver.getFieldB( );
403  EventTask eRfieldB = fieldB.asyncCommunication( __getTransactionEvent( ) );
404 
405  solver.template updateE< CORE >( currentStep );
406  __setTransactionEvent( eRfieldB );
407  solver.template updateE< BORDER >( currentStep );
408  }

416  void update_afterCurrent( uint32_t const currentStep )
417  {
418  /* These steps are the same as in the Yee solver,
419  * except the Fabsorber::ExponentialDamping::run( ) is not called,
420  * PML updates are done as part of solver.updateBHalf( ).
421  */
422  if( laserProfiles::Selected::INIT_TIME > 0.0_X )
423  LaserPhysics{ }( currentStep );
424 
425  auto & fieldE = solver.getFieldE( );
426  EventTask eRfieldE = fieldE.asyncCommunication( __getTransactionEvent( ) );
427 
428  solver.template updateBHalf< CORE >( currentStep );
429  __setTransactionEvent( eRfieldE );
430  solver.template updateBHalf< BORDER >( currentStep );
431 
432  auto & fieldB = solver.getFieldB( );
433  EventTask eRfieldB = fieldB.asyncCommunication( __getTransactionEvent( ) );
434  __setTransactionEvent( eRfieldB );
435  }
