105  HDINLINE
106  float_X
107  getMomTheta( ) const
108  {
109  //because of floating point precision x^2+y^2+z^2<y^2 for x,z<<y
110  float_X const momAbs = getMomAbs( );
111  if( momAbs * momAbs <= momentum.y( ) * momentum.y( ) )
112  return 0.0;

114  return picongpu::math::acos( momentum.y( ) * ( 1.0 / momAbs ) );
