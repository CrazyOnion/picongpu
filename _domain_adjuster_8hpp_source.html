131  void updateLocalDomainOffset( size_t const dim )
132  {
133  pmacc::GridController< simDim > & gc = pmacc::Environment< simDim >::get( ).GridController( );
134 
135  int mpiPos( gc.getPosition( )[ dim ] );
136  int numMpiRanks = gc.getGlobalSize();
137 
138  // gather mpi position in the direction we are checking
139  std::vector< int > mpiPositions( numMpiRanks );
140  MPI_CHECK( MPI_Allgather(
141  &mpiPos,
142  1,
143  MPI_INT,
144  mpiPositions.data(),
145  1,
146  MPI_INT,
147  gc.getCommunicator().getMPIComm()
148  ));
149 
150  // gather local sizes in the direction we are checking
151  std::vector< uint64_t > allLocalSizes( numMpiRanks );
152  uint64_t lSize = static_cast< uint64_t >( m_localDomainSize[ dim ] );
153  MPI_CHECK( MPI_Allgather(
154  &lSize,
155  1,
156  MPI_UINT64_T,
157  allLocalSizes.data(),
158  1,
159  MPI_UINT64_T,
160  gc.getCommunicator().getMPIComm()
161  ));
162 
163  uint64_t offset = 0u;
164  for( size_t i = 0u; i < mpiPositions.size(); ++i )
165  {
166  if( mpiPositions[ i ] < mpiPos )
167  offset += allLocalSizes[ i ];
168  }
169 
170  /* since we are not doing independent reduces per slice we need
171  * to adjust the offset result by dividing with the number of
172  * MPI ranks in all other dimensions.
173  */
174  offset /= static_cast< uint64_t >( m_numDevices.productOfComponents() / m_numDevices[ dim ] );
175  m_localDomainOffset[ dim ] = static_cast< int >( offset );
176 
177  }

184  {

235  void greaterEqualThanAbsorber( size_t const dim )
236  {
237  int validLocalSize = m_localDomainSize[ dim ];
238 
239  bool const isAbsorberEnabled = !m_isPeriodic[ dim ];
240  bool const isBoundaryDevice = ( m_mpiPosition[ dim ] == 0 || m_mpiPosition[ dim ] == m_numDevices[ dim ] - 1 );
241  if( isAbsorberEnabled && isBoundaryDevice )
242  {
243  size_t boundary = m_mpiPosition[ dim ] == 0u ? 0u : 1u;
244  int maxAbsorberCells = fields::absorber::numCells[ dim ][ boundary ];
245 
246  if( m_movingWindowEnabled && dim == 1u )
247  {
248  /* since the device changes their position during the simulation
249  * the negative and positive absorber cells must fit into the domain
250  */
251  maxAbsorberCells = static_cast< int >(
252  std::max(
253  fields::absorber::numCells[ dim ][ 0 ],
254  fields::absorber::numCells[ dim ][ 1 ]
255  )
256  );
257  }
258 
259  if( m_localDomainSize[ dim ] < maxAbsorberCells )
260  {
261  int const sCellSize = SuperCellSize::toRT()[ dim ];
262  // round up to full supercells
263  validLocalSize =
264  ( ( maxAbsorberCells + sCellSize - 1 ) / sCellSize ) *
265  sCellSize;
266  }
267 
268  if( validLocalSize != m_localDomainSize[ dim ] )
269  {

271  dim,
272  "Local grid size must be greater or equal than the largest absorber.",
273  m_localDomainSize[ dim ],
274  validLocalSize

336  void deriveGlobalDomainSize( size_t const dim )
337  {
338  uint64_t validGlobalGridSize = 0u;
339 
340  deriveLocalDomainSize( dim );
341 
342  if( m_movingWindowEnabled && dim == 1u )
343  {
344  // the local sizes in slide direction must be equal sized
345  validGlobalGridSize = static_cast< uint64_t >( m_localDomainSize[ dim ] * m_numDevices[ dim ] );
346  }
347  else
348  {
349  uint64_t localDomainSize = static_cast< uint64_t >( m_localDomainSize[ dim ] );
350  pmacc::mpi::MPIReduce mpiReduce;
351  mpiReduce(
352  pmacc::nvidia::functors::Add(),
353  &validGlobalGridSize,
354  &localDomainSize,
355  1,
356  pmacc::mpi::reduceMethods::AllReduce()
357  );
358  /* since we are not doing independent reduces per slice we need
359  * to adjust the reduce result by dividing the sizes of all other dimensions
360  * we are not check within the method call
361  */
362  validGlobalGridSize /= static_cast< uint64_t >(
363  m_numDevices.productOfComponents() / m_numDevices[ dim ]
364  );
365 
366  }
367 
368  if( m_isMaster && validGlobalGridSize != static_cast< uint64_t >( m_globalDomainSize[ dim ] ) )
369  {
370  showMessage(
371  dim,
372  "Invalid global grid size.",
373  m_globalDomainSize[ dim ],
374  validGlobalGridSize
375  );
376  }
377 
378  m_globalDomainSize[ dim ] = static_cast< int >( validGlobalGridSize );
379  }
