181  virtual void pluginLoad()
182  {
183  //fill periodic with 0
184  while (periodic.size() < 3)
185  periodic.push_back(0);
186 
187  // check on correct number of devices. fill with default value 1 for missing dimensions
188  if (devices.size() > 3)
189  {
190  std::cerr << "Invalid number of devices.\nuse [-d dx=1 dy=1 dz=1]" << std::endl;
191  }
192  else
193  while (devices.size() < 3)
194  devices.push_back(1);
195 
196  // check on correct grid size. fill with default grid size value 1 for missing 3. dimension
197  if (gridSize.size() < 2 || gridSize.size() > 3)
198  {
199  std::cerr << "Invalid or missing grid size.\nuse -g width height [depth=1]" << std::endl;
200  }
201  else
202  if (gridSize.size() == 2)
203  gridSize.push_back(1);
204 
205  if (slidingWindow && devices[1] == 1)
206  {
207  std::cerr << "Invalid configuration. Can't use moving window with one device in Y direction" << std::endl;
208  }
209 
210  DataSpace<simDim> gridSizeGlobal;
211  DataSpace<simDim> gpus;
212  DataSpace<simDim> isPeriodic;
213 
214  for (uint32_t i = 0; i < simDim; ++i)
215  {
216  gridSizeGlobal[i] = gridSize[i];
217  gpus[i] = devices[i];
218  isPeriodic[i] = periodic[i];
219  }
220 
221  Environment<simDim>::get().initDevices(gpus, isPeriodic);
222  pmacc::GridController< simDim > & gc = pmacc::Environment<simDim>::get().GridController();
223 
224  DataSpace<simDim> myGPUpos(gc.getPosition());
225 
226  if( gc.getGlobalRank() == 0 )
227  {
228  if( showVersionOnce )
229  {
230  void( getSoftwareVersions( std::cout ) );
231  }
232  }
233 
234  // calculate the number of local grid cells and
235  // the local cell offset to the global box
236  for (uint32_t dim = 0; dim < gridDistribution.size() && dim < simDim; ++dim)
237  {
238  // parse string
239  ParserGridDistribution parserGD(gridDistribution.at(dim));
240 
241  // verify number of blocks and devices in dimension match
242  parserGD.verifyDevices(gpus[dim]);
243 
244  // calculate local grid points & offset
245  gridSizeLocal[dim] = parserGD.getLocalSize(myGPUpos[dim]);
246  }
247  // by default: use an equal distributed box for all omitted params
248  for (uint32_t dim = gridDistribution.size(); dim < simDim; ++dim)
249  {
250  gridSizeLocal[dim] = gridSizeGlobal[dim] / gpus[dim];
251  }
252 
253  DataSpace<simDim> gridOffset;
254 
255  DomainAdjuster domainAdjuster(
256  gpus,
257  myGPUpos,
258  isPeriodic,
259  slidingWindow
260  );
261 
262  if(!autoAdjustGrid)
263  domainAdjuster.validateOnly();
264 
265  domainAdjuster(gridSizeGlobal, gridSizeLocal, gridOffset);
266 
267  Environment<simDim>::get().initGrids(gridSizeGlobal, gridSizeLocal, gridOffset);
268 
269  if( !slidingWindow )
270  {
271  windowMovePoint = 0.0;
272  endSlidingOnStep = 0;
273  }
274  MovingWindow::getInstance().setMovePoint(windowMovePoint);
275  MovingWindow::getInstance().setEndSlideOnStep(endSlidingOnStep);
276 
277  log<picLog::DOMAINS > ("rank %1%; localsize %2%; localoffset %3%;") %
278  myGPUpos.toString() % gridSizeLocal.toString() % gridOffset.toString();
279 
280  SimulationHelper<simDim>::pluginLoad();
281 
282  GridLayout<simDim> layout(gridSizeLocal, GuardSize::toRT() * SuperCellSize::toRT());
283  cellDescription = new MappingDesc(layout.getDataSpace(), DataSpace<simDim>(GuardSize::toRT()));
284 
285  if (gc.getGlobalRank() == 0)
286  {
287  if (MovingWindow::getInstance().isEnabled())
288  log<picLog::PHYSICS > ("Sliding Window is ON");
289  else
290  log<picLog::PHYSICS > ("Sliding Window is OFF");
291  }
292  }

317  virtual void init()
318  {
319  namespace nvmem = pmacc::nvidia::memory;
320 
321  DataConnector &dc = Environment<>::get().DataConnector();
322  initFields(dc);
323 
324  // create field solver
325  this->myFieldSolver = new fields::Solver(*cellDescription);
326 
327  // Initialize random number generator and synchrotron functions, if there are synchrotron or bremsstrahlung Photons
328  using AllSynchrotronPhotonsSpecies = typename pmacc::particles::traits::FilterByFlag<
329  VectorAllSpecies,
330  synchrotronPhotons<>
331  >::type;
332  using AllBremsstrahlungPhotonsSpecies = typename pmacc::particles::traits::FilterByFlag<
333  VectorAllSpecies,
334  bremsstrahlungPhotons<>
335  >::type;
336 
337  // create factory for the random number generator
338  const uint32_t userSeed = random::seed::ISeed< random::SeedGenerator >{}();
339  const uint32_t seed = std::hash<std::string>{}(
340  std::to_string( userSeed )
341  );
342 
343  using RNGFactory = pmacc::random::RNGProvider< simDim, random::Generator >;
344  auto rngFactory = pmacc::memory::makeUnique< RNGFactory >(
345  Environment<simDim>::get().SubGrid().getLocalDomain().size
346  );
347  if (Environment<simDim>::get().GridController().getGlobalRank() == 0)
348  {
349  log<picLog::PHYSICS >("used Random Number Generator: %1% seed: %2%") %
350  rngFactory->getName() %
351  userSeed;
352  }
353 
354  // init and share random number generator
355  pmacc::GridController<simDim>& gridCon = pmacc::Environment<simDim>::get().GridController();
356  rngFactory->init( gridCon.getScalarPosition() ^ seed );
357  dc.consume( std::move( rngFactory ) );
358 
359  // Initialize synchrotron functions, if there are synchrotron photon species
360  if(!bmpl::empty<AllSynchrotronPhotonsSpecies>::value)
361  {
362  this->synchrotronFunctions.init();
363  }
364 #if( PMACC_CUDA_ENABLED == 1 )
365  // Initialize bremsstrahlung lookup tables, if there are species containing bremsstrahlung photons
366  if(!bmpl::empty<AllBremsstrahlungPhotonsSpecies>::value)
367  {
368  meta::ForEach<
369  AllBremsstrahlungPhotonsSpecies,
370  particles::bremsstrahlung::FillScaledSpectrumMap< bmpl::_1 >
371  > fillScaledSpectrumMap;
372  fillScaledSpectrumMap(this->scaledBremsstrahlungSpectrumMap);
373 
374  this->bremsstrahlungPhotonAngle.init();
375  }
376 
377  /* Create an empty allocator. This one is resized after all exchanges
378  * for particles are created */
379  deviceHeap.reset(new DeviceHeap(0));
380 #endif
381 
382  /* Allocate helper fields for FLYlite population kinetics for atomic physics
383  * (histograms, rate matrix, etc.)
384  */
385  using AllFlyLiteIons = typename pmacc::particles::traits::FilterByFlag<
386  VectorAllSpecies,
387  populationKinetics<>
388  >::type;
389 
390  meta::ForEach<
391  AllFlyLiteIons,
392  particles::CallPopulationKineticsInit< bmpl::_1 >,
393  bmpl::_1
394  > initPopulationKinetics;
395  initPopulationKinetics(
396  gridSizeLocal
397  );
398 
399  // Allocate and initialize particle species with all left-over memory below
400  meta::ForEach< VectorAllSpecies, particles::CreateSpecies<bmpl::_1> > createSpeciesMemory;
401  createSpeciesMemory( deviceHeap, cellDescription );
402 
403  size_t freeGpuMem(0);
404  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
405  if(freeGpuMem < reservedGpuMemorySize)
406  {
407  pmacc::log< picLog::MEMORY > ("%1% MiB free memory < %2% MiB required reserved memory")
408  % (freeGpuMem / 1024 / 1024) % (reservedGpuMemorySize / 1024 / 1024) ;
409  std::stringstream msg;
410  msg << "Cannot reserve "
411  << (reservedGpuMemorySize / 1024 / 1024) << " MiB as there is only "
412  << (freeGpuMem / 1024 / 1024) << " MiB free device memory left";
413  throw std::runtime_error(msg.str());
414  }
415 
416 #if( PMACC_CUDA_ENABLED == 1 )
417  size_t heapSize = freeGpuMem - reservedGpuMemorySize;
418 
419  if( Environment<>::get().MemoryInfo().isSharedMemoryPool() )
420  {
421  heapSize /= 2;
422  log<picLog::MEMORY > ("Shared RAM between GPU and host detected - using only half of the 'device' memory.");
423  }
424  else
425  log<picLog::MEMORY > ("RAM is NOT shared between GPU and host.");
426 
427  // initializing the heap for particles
428  deviceHeap->destructiveResize(heapSize);
429  auto mallocMCBuffer = pmacc::memory::makeUnique< MallocMCBuffer<DeviceHeap> >( deviceHeap );
430  dc.consume( std::move( mallocMCBuffer ) );
431 #endif
432  meta::ForEach< VectorAllSpecies, particles::LogMemoryStatisticsForSpecies<bmpl::_1> > logMemoryStatisticsForSpecies;
433  logMemoryStatisticsForSpecies( deviceHeap );
434 
435  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
436  log<picLog::MEMORY > ("free mem after all mem is allocated %1% MiB") % (freeGpuMem / 1024 / 1024);
437 
438  IdProvider<simDim>::init();
439 
440 #if( PMACC_CUDA_ENABLED == 1 )
441  /* add CUDA streams to the StreamController for concurrent execution */
442  Environment<>::get().StreamController().addStreams(6);
443 #endif
444  }

446  virtual uint32_t fillSimulation()
447  {
448  /* assume start (restart in initialiserController might change that) */
449  uint32_t step = 0;
450 
451  /* set slideCounter properties for PIConGPU MovingWindow: assume start
452  * (restart in initialiserController might change this again)
453  */
454  MovingWindow::getInstance().setSlideCounter(0, 0);
455  /* Update MPI domain decomposition: will also update SubGrid domain
456  * information such as local offsets in y-direction
457  */
458  GridController<simDim> &gc = Environment<simDim>::get().GridController();
459  gc.setStateAfterSlides(0);
460 
461  DataConnector &dc = Environment<>::get().DataConnector();
462  auto fieldE = dc.get< FieldE >( FieldE::getName(), true );
463  auto fieldB = dc.get< FieldB >( FieldB::getName(), true );
464 
465  /* fill all objects registed in DataConnector */
466  if (initialiserController)
467  {
468  initialiserController->printInformation();
469  if (this->restartRequested)
470  {
471  /* we do not require '--checkpoint.restart.step' if a master checkpoint file is found */
472  if (this->restartStep < 0)
473  {
474  std::vector<uint32_t> checkpoints = readCheckpointMasterFile();
475 
476  if (checkpoints.empty())
477  {
478  throw std::runtime_error(
479  "Restart failed. You must provide the '--checkpoint.restart.step' argument. See picongpu --help."
480  );
481  } else
482  this->restartStep = checkpoints.back();
483  }
484 
485  initialiserController->restart((uint32_t)this->restartStep, this->restartDirectory);
486  step = this->restartStep;
487 
495  cellwiseOperation::CellwiseOperation< GUARD > guardBGField( *cellDescription );
496  namespace nvfct = pmacc::nvidia::functors;
497  guardBGField( fieldE, nvfct::Add(), FieldBackgroundE( fieldE->getUnit() ),
498  step, FieldBackgroundE::InfluenceParticlePusher );
499  guardBGField( fieldB, nvfct::Add(), FieldBackgroundB( fieldB->getUnit() ),
500  step, FieldBackgroundB::InfluenceParticlePusher );
501  }
502  else
503  {
504  initialiserController->init();
505  meta::ForEach< particles::InitPipeline, particles::CallFunctor<bmpl::_1> > initSpecies;
506  initSpecies( step );
507  }
508  }
509 
510  size_t freeGpuMem(0u);
511  Environment<>::get().MemoryInfo().getMemoryInfo(&freeGpuMem);
512  log<picLog::MEMORY > ("free mem after all particles are initialized %1% MiB") % (freeGpuMem / 1024 / 1024);
513 
514  // generate valid GUARDS (overwrite)
515  EventTask eRfieldE = fieldE->asyncCommunication(__getTransactionEvent());
516  __setTransactionEvent(eRfieldE);
517  EventTask eRfieldB = fieldB->asyncCommunication(__getTransactionEvent());
518  __setTransactionEvent(eRfieldB);
519 
520  dc.releaseData( FieldE::getName() );
521  dc.releaseData( FieldB::getName() );
522 
523  return step;
524  }

666  void initFields( DataConnector& dataConnector )
667  {
668  using pmacc::memory::makeUnique;
669  auto fieldB = makeUnique< FieldB >( *cellDescription );
670  dataConnector.consume( std::move( fieldB ) );
671  auto fieldE = makeUnique< FieldE >( *cellDescription );
672  dataConnector.consume( std::move( fieldE ) );
673  auto fieldJ = makeUnique< FieldJ >( *cellDescription );
674  dataConnector.consume( std::move( fieldJ ) );
