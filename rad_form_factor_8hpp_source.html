57  HDINLINE float_X operator()( const float_X N, const float_X omega, vector_X const & observer_unit_vec ) const
58  {
59  float_X sincValue = float_X( 1.0 );
60  for( uint32_t d = 0; d < DIM3; ++d )
61  sincValue *= math::sinc( observer_unit_vec[d] * cellSize[d] / ( SPEED_OF_LIGHT * float_X( 2.0 ) ) * omega );
62 
63  // here we combine sinc^2(..) with (...)^T_shapeOrder to ...^(2 * T_shapeOrder)
64  return math::sqrt( N + ( N * N - N ) * util::pow( sincValue , 2 * T_shapeOrder ) );
65  }
