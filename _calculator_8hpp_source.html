118  float_X
119  calcEnergyPerp( ) const
120  {
121  // a, x and y are temporary variables without an explicit physical meaning
122  float_X const a = uSquared * parMomCosTheta * parMomSinTheta *
123  parMomSinPhi * detectorCosTheta;
124 
125  // Denominator
126  float_X const x = parSqrtOnePlusUSquared -
127  particle.getU( ) * parMomSinTheta * parMomCosPhi * detectorSinTheta;
128  float_X const y = particle.getU( ) * parMomCosTheta * detectorCosTheta;
129 
130  float_X denominator = x * x - y * y;
131 
132  // Preventing division by 0
133  if( math::abs( denominator ) < DIV_BY_ZERO_MINIMUM )
134  {
135  if( denominator < 0.0 )
136  denominator = -DIV_BY_ZERO_MINIMUM;
137  else
138  denominator = DIV_BY_ZERO_MINIMUM;
139  }
140 
141  return a / denominator;
142  }

155  HDINLINE
156  float_X
157  calcEnergyPara( ) const
158  {
159  // a, b, c, x and y are just temporary variables without an explicit physical meaning
160  float_X const a = particle.getU( ) * parMomCosTheta;
161  float_X const b = particle.getU( ) * parMomSinTheta * parMomCosPhi;
162  float_X const c = parSqrtOnePlusUSquared * detectorSinTheta;
163 
164  // Denominator
165  float_X const x = parSqrtOnePlusUSquared -
166  particle.getU( ) * parMomSinTheta * parMomCosPhi * detectorSinTheta;
167  float_X const y = particle.getU( ) * parMomCosTheta * detectorCosTheta;
168 
169  float_X denominator = x * x - y * y;
170 
171  // Preventing division by 0
172  if( math::abs( denominator ) < DIV_BY_ZERO_MINIMUM )
173  {
174  if( denominator < 0.0 )
175  denominator = -DIV_BY_ZERO_MINIMUM;
176  else
177  denominator = DIV_BY_ZERO_MINIMUM;
178  }
179 
180  return a * ( b - c ) / denominator;
181  }

192  HDINLINE
193  complex_X
194  calcFormFactorExponent( ) const
195  {
196  // If case for longitudinal moving particles... leads to 0 later in the kernel
197  if ( math::abs( parMomCosTheta ) <= DIV_BY_ZERO_MINIMUM )
198  return complex_X( -1.0, 0.0 );
199 
200  float_X const a = detectorSinTheta * parMomSinTheta * math::cos( parMomPhi - detectorPhi );
201  float_X const b = - ( particle.getPosPara( ) ) * ( 1 / particle.getVel( ) - a / SPEED_OF_LIGHT) / ( parMomCosTheta );
202  float_X const c = - detectorSinTheta * particle.getPosPerp( ) * math::cos( particle.getPosPhi( ) - detectorPhi );
203 
204  complex_X const fpara = complex_X( 0.0, b );
205  complex_X const fperp = complex_X( 0.0, c );
206  return fpara + fperp;
